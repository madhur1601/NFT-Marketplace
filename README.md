# Clone this repo

NOTE: Change the port 8545 to 7545 in 'truffle-config.js' (As per the Ganache)


$ cd NFT-Marketplace

# install the dependencies
$ yarn

# run ganache
$ ganache-cli

# deploy de contracts on the blockchain
$ truffle migrate

# run the client-side
$ cd client
$ yarn
$ yarn start

# run the backend
$ cd backend
$ yarn
$ yarn start
